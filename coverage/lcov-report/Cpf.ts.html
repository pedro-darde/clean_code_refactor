export class CPF {
  VALID_LENGTH = 11;
  FIRST_VERIFIER_DIGIT_FACTOR = 10;
  SECOND_VERIFIER_DIGIT_FACTOR = 11;
 
  private cpf: string;
  constructor(cpf: string) {
    this.cpf = cpf;
  }
 
  isValid() {
    if (!this.cpf) return false;
    this.removeSpecialCharactersFromCPF();
    if (!this.hasCorrectLength()) return false;
    if (this.isAllTheSameCharOnCpf()) return false;
 
    const digit1 = this.getFactorDigit(this.FIRST_VERIFIER_DIGIT_FACTOR);
    const digit2 = this.getFactorDigit(this.SECOND_VERIFIER_DIGIT_FACTOR);
    let checkDigit = this.extractCheckDigit();
    const calculatedCheckDigit = `${digit1}${digit2}`;
    return checkDigit == calculatedCheckDigit;
  }
 
  private removeSpecialCharactersFromCPF() {
    this.cpf = this.cpf.replace(/\D+/g, "");
  }
 
  private hasCorrectLength() {
    return this.cpf.length === this.VALID_LENGTH;
  }
 
  private isAllTheSameCharOnCpf() {
    const [firstDigit, ...restDigits] = this.cpf;
    return restDigits.every((digit) => digit === firstDigit);
  }
 
  private getFactorDigit(factor: number) {
    let total = 0;
		for (const digit of this.cpf) {
			if (factor > 1) {
				total += parseInt(digit) * factor--;
			}
		}
		const rest = total%11;
		return (rest < 2) ? 0 : 11 - rest;
  }
 
  extractCheckDigit() {
    return this.cpf.slice(-2);
  }
}
